\def\@maketitle{%

  \begin{center}%
    \let \footnote \thanks
    {\Large\semib{\textssc{\@title}}\par}
        \vskip 0.1\baselineskip
    \begin{tabular}[t]{c}%

        \@date
            \end{center}%

}
